# ============================================================
# Deploy requirements -- Hugging Face Spaces (Docker SDK), Groq mode
# ============================================================
# Only what the Groq/API generation path needs. NO torch, NO transformers,
# NO bitsandbytes, NO unsloth: in LLM_PROVIDER=groq mode the local model is
# never loaded (see src/rag/rag_pipeline.py), so those heavy deps are skipped
# to keep the image small.
#
# NOTE: sentence-transformers is still required -- retrieval embeds the query
# clause locally with all-MiniLM-L6-v2 (CPU). It will pull a CPU build of torch
# transitively; that is the embedder's dependency, not the LLM's. If you want a
# torch-free image, swap the embedder for an API/ONNX embedding backend.

# Web API
# fastapi and starlette are pinned EXACTLY, not just floored, because newer fastapi
# releases (>=0.130) drop the starlette<1.0.0 ceiling and can resolve to starlette
# 1.x -- which conflicts with prometheus-fastapi-instrumentator 7.x below (that package
# requires starlette<1.0.0). Pinning here removes that resolver ambiguity entirely.
fastapi==0.129.0
starlette==0.52.1
uvicorn[standard]>=0.29.0
pydantic>=2.4.0

# Observability (Prometheus /metrics; pulls in prometheus-client).
# Pinned to the 7.x line (not >=7.0.0) because 8.x requires starlette>=1.0.0, which
# would conflict with the fastapi/starlette pins above. 7.1.0 requires
# starlette>=0.30.0,<1.0.0 -- compatible with fastapi 0.129.0's own
# starlette>=0.40.0,<1.0.0 constraint.
prometheus-fastapi-instrumentator==7.1.0

# Remote generation (Groq)
groq>=0.11.0
python-dotenv>=1.0.0

# RAG retrieval (embedding + vector store) -- runs on CPU
sentence-transformers>=3.0.0
chromadb>=0.5.0
